private async Task EnsureFolderExistsAsync(
    string driveId,
    string folderName)
{
    try
    {
        await _graphClient
            .Drives[driveId]
            .Root
            .ItemWithPath(folderName)
            .GetAsync();
    }
    catch
    {
        await _graphClient
            .Drives[driveId]
            .Root
            .Children
            .PostAsync(new DriveItem
            {
                Name = folderName,
                Folder = new Folder()
            });
    }
}
