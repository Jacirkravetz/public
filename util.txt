public DataIntegrationDashboardForm()
{
    InitializeComponent();
    this.Load += DataIntegrationDashboardForm_Load;
}

private void DataIntegrationDashboardForm_Load(object? sender, EventArgs e)
{
    // valores desejados mínimos
    int panel1Min = 220;
    int panel2MinDesired = 600;

    // largura disponível do cliente
    int totalWidth = this.ClientSize.Width;

    // garante que panel2Min não seja maior que o que cabe
    int panel2Min = panel2MinDesired;
    if (totalWidth < panel1Min + panel2MinDesired + 10) // +10 margem
    {
        // reduz conservadoramente o panel2Min para caber
        panel2Min = Math.Max(200, totalWidth - panel1Min - 10);
    }

    // applique os mínimos e ajuste SplitterDistance para ficar consistente
    this.mainSplit.Panel1MinSize = panel1Min;
    this.mainSplit.Panel2MinSize = panel2Min;

    // define SplitterDistance em um valor válido (por exemplo um pouco maior que panel1Min)
    int desiredSplitter = Math.Min(Math.Max(panel1Min + 40, this.mainSplit.SplitterDistance), totalWidth - panel2Min - 10);
    // Math.Clamp alternativa (se não disponível):
    if (desiredSplitter < panel1Min) desiredSplitter = panel1Min;
    if (desiredSplitter > totalWidth - panel2Min - 10) desiredSplitter = totalWidth - panel2Min - 10;

    this.mainSplit.SplitterDistance = desiredSplitter;
}
