public async Task<string?> GetAccessTokenAsync()
        {
            try
            {
                var scopes = new[] { "https://graph.microsoft.com/.default" }; // Defina os escopos da API
                string token = await _tokenAcquisition.GetAccessTokenForUserAsync(scopes);
                return token;
            }
            catch (Exception ex)
            {
                Console.WriteLine($"Erro ao obter o token: {ex.Message}");
                return null;
            }
        }
