DECLARE
    v_count NUMBER;
BEGIN
    -- Verifica se a tabela temporária já existe no banco de dados
    SELECT COUNT(*) INTO v_count 
    FROM ALL_TABLES 
    WHERE TABLE_NAME = 'TMP_CLIENTES_SINACOR' AND OWNER = 'SEU_ESQUEMA';

    -- Se a tabela não existir, cria
    IF v_count = 0 THEN
        EXECUTE IMMEDIATE 'CREATE GLOBAL TEMPORARY TABLE TMP_CLIENTES_SINACOR (
            CNPJ VARCHAR2(20),
            CONTA VARCHAR2(20),
            STATUS VARCHAR2(10)
        ) ON COMMIT PRESERVE ROWS';
        
        DBMS_OUTPUT.PUT_LINE('Tabela TMP_CLIENTES_SINACOR criada.');
    ELSE
        DBMS_OUTPUT.PUT_LINE('Tabela TMP_CLIENTES_SINACOR já existe.');
    END IF;
END;
/
