private string GetLogClass(LogLevel level) => level switch
{
    LogLevel.Trace => "log-trace",          // 🔍 Logs detalhados para debugging
    LogLevel.Debug => "log-debug",          // 🐞 Logs para depuração
    LogLevel.Information => "log-info",     // ℹ️ Informações gerais
    LogLevel.Warning => "log-warning",      // ⚠️ Avisos e possíveis problemas
    LogLevel.Error => "log-error",          // ❌ Erros que impactam a funcionalidade
    LogLevel.Critical => "log-critical",    // 🚨 Falhas graves no sistema
    LogLevel.None => "log-none",            // 🚫 Nenhuma categoria específica
    _ => "log-default"                      // 🔄 Caso algum novo nível seja adicionado futuramente
};
