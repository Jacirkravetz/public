DataConvertida = 
VAR TextoData = [SuaColunaDeTexto]
VAR Dia = VALUE(LEFT(TextoData, 2))
VAR Mes = VALUE(MID(TextoData, 4, 2))
VAR Ano = VALUE(RIGHT(TextoData, 4))

RETURN
IF(
    ISBLANK(Dia) || ISBLANK(Mes) || ISBLANK(Ano) ||
    Dia > 31 || Mes > 12 || Ano < 1900 || YEAR(TODAY()) + 100 < Ano,
    BLANK(),
    DATE(Ano, Mes, Dia)
)
