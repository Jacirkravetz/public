sessionManager.MessageReceived += (message, sessionID) =>
{
    // Chama o método async, mas não bloqueia o evento
    _ = HandleMessageAsync(message, sessionID);
};

async Task HandleMessageAsync(Message message, SessionID sessionID)
{
    try
    {
        Console.WriteLine($"Received message from session {sessionID}: {message}");
        await PushAsync(message, sessionID);
    }
    catch (Exception ex)
    {
        Console.WriteLine($"Error processing message: {ex}");
    }
}
