public void Execute(CancellationToken cancellationToken)
{
    _engineBootstrapper.Start(fixMessage =>
    {
        // Se o token foi cancelado, interrompe o processamento
        if (cancellationToken.IsCancellationRequested)
        {
            // Opcional: aqui você pode lançar uma exceção, ou simplesmente retornar para interromper o callback
            return;
        }

        var tabularMessage = ConvertFixToTabular(fixMessage);
        _processor.Push(tabularMessage);

    }, cancellationToken);
}
