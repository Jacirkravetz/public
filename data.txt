using System;

public class CutOffWindow
{
    public Guid Id { get; set; }                      // Chave primária
    public Guid StepId { get; set; }                  // FK para o Step do tipo "CutOffControl"

    public DayOfWeek DayOfWeek { get; set; }          // 0 = Domingo, 6 = Sábado

    public TimeSpan StartTime { get; set; }           // Ex: 08:00
    public TimeSpan EndTime { get; set; }             // Ex: 12:00

    public bool ExcludeHolidays { get; set; }         // Se deve ignorar feriados
}
