ConvertedDate =
VAR TextDate = [YourTextColumn]
RETURN
IF (
    LEN(TextDate) = 10
        && MID(TextDate, 3, 1) = "/"
        && MID(TextDate, 6, 1) = "/"
        && ISNUMBER(VALUE(LEFT(TextDate, 2)))
        && ISNUMBER(VALUE(MID(TextDate, 4, 2)))
        && ISNUMBER(VALUE(RIGHT(TextDate, 4))),
    DATE(
        VALUE(RIGHT(TextDate, 4)),     -- Year
        VALUE(MID(TextDate, 4, 2)),    -- Month
        VALUE(LEFT(TextDate, 2))       -- Day
    ),
    BLANK()
)
