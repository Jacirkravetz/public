private static string GetDayOfWeekDescriptions()
{
    return string.Join(", ",
        Enum.GetValues(typeof(DayOfWeek))
            .Cast<DayOfWeek>()
            .Select(d => $"{(int)d} ({d})"));
}
RuleFor(x => x.DayOfWeek)
    .Must(day => Enum.IsDefined(typeof(DayOfWeek), day))
    .WithMessage($"DayOfWeek must be one of: {GetDayOfWeekDescriptions()}");



 RuleFor(x => x.StartTime)
            .LessThan(x => x.EndTime)
            .WithMessage("StartTime must be earlier than EndTime.");

        RuleFor(x => x.EndTime)
            .GreaterThan(x => x.StartTime)
            .WithMessage("EndTime must be later than StartTime.");
