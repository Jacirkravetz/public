public class StringDeserializer : IDeserializer<object>
{
    public object Deserialize(ReadOnlySpan<byte> data, bool isNull, SerializationContext context)
    {
        return isNull ? null : System.Text.Encoding.UTF8.GetString(data);
    }
}


   SchemaType.String => new ConsumerBuilder<string, object>(consumerConfig)
                .SetKeyDeserializer(Deserializers.Utf8)
                .SetValueDeserializer(new StringDeserializer())
                .Build(),
